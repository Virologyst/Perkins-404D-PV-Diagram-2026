        private void button1_Click_1(object sender, EventArgs e)
        {
            // Convert listA and listB to double arrays
            double[] x = listX.ToArray();
            double[] y = listY.ToArray();

            // Create a filename with date and time separated
            string filename = DateTime.Now.ToString("yyyy-MM-dd_HH-mm-ss") + ".csv";

            using (SaveFileDialog saveFileDialog = new SaveFileDialog())
            {
                saveFileDialog.Filter = "CSV (*.csv|*.csv|All files (*.*)|*.*";
                saveFileDialog.Title = "Save Chart Data";
                saveFileDialog.FileName = filename;

                if (saveFileDialog.ShowDialog() == DialogResult.OK)
                {
                    try
                    {
                        using (StreamWriter writer = new StreamWriter(saveFileDialog.FileName))
                        {
                            for (int i = 0; i < x.Length; i++)
                            {
                                writer.WriteLine($"{x[i]},{y[i]}");
                            }
                        }
                        MessageBox.Show("Data saved successfully.", "Save", MessageBoxButtons.OK, MessageBoxIcon.Information);
                    }
                    catch (Exception ex)
                    {
                        MessageBox.Show("Error saving data: " + ex.Message);
                    }
                }
            }
        }
